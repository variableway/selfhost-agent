## Technical Notes

```rust
#[tauri::command]
async fn detect_platform() -> PlatformInfo {
    PlatformInfo {
        os: std::env::consts::OS.into(),  // "macos" | "windows" | "linux"
        arch: std::env::consts::ARCH.into(),
        shell: detect_default_shell(),
    }
}

#[tauri::command]
async fn check_tools(tools: Vec<String>) -> HashMap<String, ToolInfo> {
    let mut result = HashMap::new();
    for tool in tools {
        match which_tool(&tool) {
            Ok(path) => {
                let version = get_version(&tool).await;
                result.insert(tool, ToolInfo { installed: true, version, path });
            },
            Err(_) => {
                result.insert(tool, ToolInfo { installed: false, version: None, path: None });
            }
        }
    }
    result
}
```
